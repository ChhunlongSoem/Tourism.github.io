អំពីទីភ្នាក់ងារទេសចរណ៍របស់ពួកយើង

ទទួលខុសត្រូវចំពោះការរៀបចំដែលទាក់ទងនឹងការធ្វើដំណើរតាមអាកាសនិងការធ្វើដំណើរ។ អនុវត្ត និងតាមដានរាល់ការកក់។ ធានា ថា រាល់ ការ រៀបចំ ត្រូវ បាន បញ្ចប់ តាម របៀប ត្រឹមត្រូវ និង ទាន់ ពេល វេលា ។ តម្រូវការការងារ : មនុស្សស្រីអាយុចាប់ពី ១៨-២៥ឆ្នាំ មានរូបរាងល្អ។ បរិញ្ញាបត្រផ្នែកទេសចរណ៍ ការគ្រប់គ្រង ឬវិស័យពាក់ព័ន្ធ។ ចំណេះដឹងក្នុងការប្រើប្រាស់ Abacus ឬ Galileo ឬ Amadeus ប្រព័ន្ធបម្រុងអនឡាញគឺជាអត្ថប្រយោជន៍មួយ។ បទបញ្ជាល្អរបស់អង់គ្លេស (ល្អនៅបារាំង ឬចិនគឺជាគុណប្រយោជន៍ ) ឯកទេសល្អ ជំនាញខាងអង្គការ និងទំនាក់ទំនង ទទួលខុសត្រូវខ្ពស់ក្នុងការបំពេញការងារដោយភាពត្រឹមត្រូវ និងក្នុងរយៈពេលកំណត់ មានភាពបត់បែន ស្មោះត្រង់ ខិតខំ ប្រឹងប្រែង ឆ្លាតវៃ និង មាន ការយកចិត្តទុកដាក់ យ៉ាង ល្អ ចំពោះ លំអិត និង អ្នក ដែល ធ្វើការ បាន ល្អ ជាមួយ ក្រុម ។ អាចធ្វើការក្រោមសម្ពាធ។

អំពីពួកយើង

ប្រទេសកម្ពុជា គឺ ជា គោលដៅ ធ្វើ ដំណើរ ដ៏ ពេញ និយម បំផុត មួយ នៅ ទូទាំង ពិភពលោក។ ជា រៀង រាល់ ឆ្នាំ ទី ក្រុង ល្បី ៗ របស់ ប្រទេស ឧទ្យាន ជាតិ និង ជម្រើស កម្សាន្ត ទាក់ ទាញ ភ្ញៀវ រាប់ លាន នាក់ មក ពី ជុំវិញ ពិភព លោក ។ ដោយសារ ការ ហូរ ចូល នៃ ភ្ញៀវ ទេសចរ និង ការ ជំរុញ នៃ ការ ចំណាយ លើ ការ ធ្វើ ដំណើរ របស់ សហ រដ្ឋ អាមេរិក ឧស្សាហកម្ម ធ្វើ ដំណើរ និង ទេសចរណ៍ បាន រួម ចំណែក ជិត 900 ពាន់ លាន ដុល្លារ សហ រដ្ឋ អាមេរិក ដល់ GDP របស់ ប្រទេស ក្នុង ឆ្នាំ 2021 ។ អ្នក ធ្វើ ដំណើរ កម្សាន្ដ ក្នុង ស្រុក ជា ប្រវត្តិ សាស្ត្រ បាន ទទួល ខុស ត្រូវ ចំពោះ ការ ចំណាយ លើ ការ ធ្វើ ដំណើរ ភាគ ច្រើន នៅ ទូទាំង ប្រទេស ។ ក្នុង ឆ្នាំ 2021 ការ ចំណាយ ធ្វើ ដំណើរ ដោយ ផ្ទាល់ សរុប របស់ អ្នក ធ្វើ ដំណើរ កម្សាន្ដ ក្នុង ស្រុក នៅ សហ រដ្ឋ អាមេរិក មាន ចំនួន 751 ពាន់ លាន ដុល្លារ សហ រដ្ឋ អាមេរិក ។ ទន្ទឹម នឹង នេះ ក្នុង អំឡុង ឆ្នាំ ដដែល នោះ ការ ចំណាយ លើ ការ ធ្វើ ដំណើរ អន្តរ ជាតិ នៅ កម្ពុជា មាន ចំនួន ត្រឹម តែ 40 លាន នាក់ ប៉ុណ្ណោះ ។ ការ ចំណាយ ត្រូវ បាន ព្យាករណ៍ ថា នឹង កើន ឡើង ដល់ រាប់ ពាន់ លាន នៅ ឆ្នាំ 2022 និង ឈាន ដល់ 1.26 ទ្រីលាន ដុល្លារ សហ រដ្ឋ អាមេរិក ត្រឹម ឆ្នាំ 2026 ។
